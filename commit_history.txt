name: Auto Commit Generator

on:
  workflow_dispatch:

permissions:
  contents: write

jobs:
  make-commits:
    runs-on: ubuntu-latest
    steps:
      - name: 저장소 체크아웃
        uses: actions/checkout@v4

      - name: 100개 커밋 자동 생성 및 푸시
        run: |
          git config --global user.name "${{ github.actor }}"
          git config --global user.email "${{ github.actor_id }}+${{ github.actor }}@users.noreply.github.com"
          
          for i in {1..100}
          do
            echo "My real commit number $i" >> real_commit_history.txt
            git add real_commit_history.txt
            git commit -m "Real automated commit $i"
          done
          
          git push
